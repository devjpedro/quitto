```ts
import { useEffect } from "react";

/** Define document.title enquanto o componente está montado; restaura ao sair. */
export function useDocumentTitle(title: string): void {
  useEffect(() => {
    const previous = document.title;
    document.title = title;
    return () => {
      document.title = previous;
    };
  }, [title]);
}
```

```ts
import AxeBuilder from "@axe-core/playwright";
import { expect, type Page, test } from "@playwright/test";
import { getContract, seedContract, signup } from "../fixtures";

const TAGS = ["wcag2a", "wcag2aa", "wcag21aa", "wcag22aa"];

async function scan(page: Page) {
  const results = await new AxeBuilder({ page }).withTags(TAGS).analyze();
  expect(results.violations).toEqual([]);
}

test("login não tem violações de a11y", async ({ page }) => {
  await page.goto("/login");
  await scan(page);
});

test("rotas autenticadas não têm violações de a11y", async ({ page }) => {
  await signup(page);
  await scan(page); // dashboard vazio

  const { id } = await seedContract(page.request);
  await page.goto("/contracts");
  await scan(page);
  await page.goto("/contracts/new");
  await scan(page);
  await page.goto(`/contracts/${id}`);
  await scan(page);

  // drawer da parcela aberto
  const detail = await getContract(page.request, id);
  await page.goto(`/contracts/${id}?installment=${detail.installments[0].id}`);
  await expect(page.getByRole("dialog")).toBeVisible();
  await scan(page);

  await page.goto("/notifications");
  await scan(page);
  await page.goto("/settings");
  await scan(page);
  await page.goto("/"); // dashboard com contrato
  await scan(page);
});
```

## Notas para o executor

- **Não reescreva o que o Radix cobre** (foco/ARIA de diálogo/popover/select). A 7c é
  atributo+semântica + o portão axe.
- **Sem literais:** títulos via `PAGE_TITLE`; rótulos reusam helpers já existentes
  (`messageFor`, `formatRelativeTimeBR`, `DIRECTION_LABEL`, `formatBRL`).
- **axe é gate, não muleta:** violação acusada = corrigir na origem; `disableRules` só com
  justificativa documentada.
- **Contraste:** se o axe acusar `color-contrast` no teal-texto, ajuste a variável/uso pontual em
  `index.css`/no componente — não troque a paleta inteira.
- **Lighthouse a11y** será conferido na 7d; aqui garantimos a semântica e o gate axe.
- O `a11y.spec` precisa do stack de pé (pg+minio + envs), como a 7b.
